<script type="text/javascript">
   function refresh(){
      var p=prov.value;
      if(p==""){
         city.length=0;
         city.options.add(new Option("--请选择城市--"))
      }
      else{
        var url="list";
        var params="proc="+prov.value ;
        sendRequest(url,params,'POST',show);
      }
   }
   function show(){
      if (httpRequest.readyState == 4) { 
         if (httpRequest.status == 200) {
            var citylist=httpRequest.responseText.split(",");
            var citynum=citylist.length;
            city.length=0;
            for(i=0;i<citynum;i++)
               city.options.add(new Option(citylist[i]))
            }
       }
   }
</script>

       <select name="prov" onchange="refresh();">
